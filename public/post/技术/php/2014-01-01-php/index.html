##查找文件下所有文件名

$str = '';
$dir = "../Tpl/";
$preg2 = '/[A-Za-z]/';
if (is_dir ( $dir )) {
	if ($dh = opendir ( $dir )) {
		while ( ($file = readdir ( $dh )) !== false ) {
			if (preg_match ( $preg2, $file )) {
				$zt = '';
				if ($info ['theme'] == $file) {
					$zt = 'selected';
				}
				$str .= '<option value=' . $file . ' ' . $zt . '>' . $file . '</option>';
			}
		}
		closedir ( $dh );
	}
}
